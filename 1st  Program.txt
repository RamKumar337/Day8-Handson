package day8handson;

import java.util.*;

class Average
{
	static int sum = 0;
	public static void sum(List<Integer> l) {
	    int sum = 0;
	    double avr;
	    for (int i: l) {
	        sum += i;
	    }
	    System.out.println("Total Runs: "+sum);
	    System.out.println("The Average Score is: "+ (avr= sum / l.size()));
	}
	 
	
}
