package Day9handson;

import java.util.*;

public class example5 {

	public static void main(String[] args) {
		List<String> venue=new ArrayList<>();
		Scanner in= new Scanner(System.in);
		int n=in.nextInt();
		for(int i=0;i<=n;i++)
			venue.add(in.nextLine());
		System.out.println("Enter venue to find frequence: ");
		String venue1=in.nextLine();
		int freq= Collections.frequency(venue, venue1);
		System.out.println("venue frequence is: "+freq);
		
	
	}

}
