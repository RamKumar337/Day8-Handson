package day8handson;

import java.util.*;
import java.util.stream.Collectors;


 class example1  {

	public static void main(String[] args) {
		Scanner in  =new Scanner(System.in);
	
		System.out.println("Enter the num of matches: ");
		int n=in.nextInt();
		System.out.println("Enter the Runs scored "+n+" IPL Matches");
		List<Integer> l=new ArrayList<>();
		for(int i=0;i<n;i++)
			l.add(in.nextInt());
		Collection<Integer> sl = l.stream().sorted().collect(Collectors.toList());
		System.out.println("Sorted number");
		for(int sl1: sl)
		{
			System.out.println(sl1);
		}
			
			
		
		

	}


}
